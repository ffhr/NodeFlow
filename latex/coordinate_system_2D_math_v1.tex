\documentclass[border=1cm]{standalone} % Standalone document with equal borders
\usepackage{tikz}

\begin{document}

\begin{tikzpicture}[scale=1.0]
    % Define range for the grid
    \def\xmax{16.5} % Ensuring equal space
    \def\ymax{16.5} % Ensuring equal space

    % Expand bounding box to ensure equal margins on all sides
    \useasboundingbox (-\xmax-2,-\ymax-2) rectangle (\xmax+2,\ymax+2);

    % Draw X and Y axes
    \draw[->] (-\xmax-1,0) -- (\xmax+1,0) node[right] {\Huge $x$}; % Increased label size
    \draw[->] (0,-\ymax-1) -- (0,\ymax+1) node[above] {\Huge $y$}; % Increased label size

    % Draw grid dots with more density
    \foreach \x in {-15,-12.5,-10,-7.5,-5,-2.5,0,2.5,5,7.5,10,12.5,15} {
        \foreach \y in {-15,-12.5,-10,-7.5,-5,-2.5,0,2.5,5,7.5,10,12.5,15} {
            \fill (\x,\y) circle (1pt);
        }
    }
\end{tikzpicture}

\end{document}
